            cmd.ExecuteNonQuery();
            SqlDataReader dr;
            dr = cmd.ExecuteReader();

                txtType.Text = Type;

                string Name = (string)dr["الأسم"].ToString();
                Nametxt.Text = Name;

                string CustodyVariable = (string)dr["العهدة"].ToString();
                custodytxt.Text = CustodyVariable;

                txtDept.Text = Dept;

            dr.Close();
